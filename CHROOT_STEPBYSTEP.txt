#Step 2: Copy and extract the Starfish offline bundle

#Step 3: Mount filesystems and enter the chroot

mount --bind /proc /opt/rocky-chroot/proc
mount --bind /sys /opt/rocky-chroot/sys
chroot /opt/rocky-chroot /bin/bash
#

#Step 4: Install the systemctl wrapper
#This wrapper intercepts PostgreSQL-related systemctl calls and translates them to commands that work without systemd:

mv /usr/bin/systemctl /usr/bin/systemctl.real
cat > /usr/bin/systemctl << 'EOF'
#!/bin/bash

echo "DEBUG: systemctl called with: $*" | tee -a /tmp/systemctl.log >&2

# PostgreSQL handlers
if [[ "$1" == "show" && "$*" == *"postgresql-17"* ]]; then
    grep "Environment=PGDATA" /etc/systemd/system/postgresql-17.service.d/01_wants-network-online_kill-mode-process.conf 2>/dev/null
    exit 0
fi

if [[ "$1" == "start" && "$2" == *"postgresql-17"* ]]; then
    PGDATA=$(grep "Environment=PGDATA" /etc/systemd/system/postgresql-17.service.d/01_wants-network-online_kill-mode-process.conf | cut -d= -f3)
    su - postgres -c "/usr/pgsql-17/bin/pg_ctl start -D ${PGDATA} -l ${PGDATA}/startup.log -w"
    exit $?
fi

if [[ "$1" == "stop" && "$2" == *"postgresql-17"* ]]; then
    PGDATA=$(grep "Environment=PGDATA" /etc/systemd/system/postgresql-17.service.d/01_wants-network-online_kill-mode-process.conf | cut -d= -f3)
    su - postgres -c "/usr/pgsql-17/bin/pg_ctl stop -D ${PGDATA} -m fast"
    exit $?
fi

# Starfish services handler
if [[ "$1" == "start" || "$1" == "restart" || "$1" == "stop" ]]; then
    ACTION="$1"
    SERVICE="$2"
    
    # Handle starfish.target - start all wanted services
    if [[ "$SERVICE" == "starfish.target" ]]; then
        for svc in sf-auth sf-auth-v2 sf-archive sf-config sf-cron sf-dispatcher sf-gateway sf-loki29 sf-nginx2 sf-pgagent sf-pgloader sf-scans sf-tagsets-v2 sf-temp sf-volumes sf-workflows; do
            $0 "$ACTION" "$svc"
        done
        exit 0
    fi
    
    # Handle sf-agent
    if [[ "$SERVICE" == "sf-agent" || "$SERVICE" == "sf-agent.service" ]]; then
        case "$ACTION" in
            start|restart)
                /opt/starfish/bin/agent start >>/opt/starfish/log/agent.out 2>&1
                ;;
            stop)
                /opt/starfish/bin/agent stop >>/opt/starfish/log/agent.out 2>&1
                ;;
        esac
        exit $?
    fi
    
    # Handle other sf-* services - they follow same pattern
    if [[ "$SERVICE" == sf-* ]]; then
        SVC_NAME="${SERVICE%.service}"
        SVC_NAME="${SVC_NAME#sf-}"
        if [[ -x "/opt/starfish/bin/${SVC_NAME}" ]]; then
            case "$ACTION" in
                start|restart)
                    /opt/starfish/bin/${SVC_NAME} start >>/opt/starfish/log/${SVC_NAME}.out 2>&1
                    ;;
                stop)
                    /opt/starfish/bin/${SVC_NAME} stop >>/opt/starfish/log/${SVC_NAME}.out 2>&1
                    ;;
            esac
            exit $?
        fi
    fi
fi

if [[ "$1" == "daemon-reload" || "$1" == "enable" ]]; then
    exit 0
fi

if [[ "$1" == "is-enabled" && "$2" == *"postgresql"* ]]; then
    echo "enabled"
    exit 0
fi

/usr/bin/systemctl.real "$@"
EOF
chmod +x /usr/bin/systemctl



#Step 5: Run the installer
